17//! An attribute to create an atomic wrapper around a C-style enum.
18//!
19//! Internally, the generated wrapper uses an `AtomicUsize` to store the value.
20//! The atomic operations have the same semantics as the equivalent operations
21//! of `AtomicUsize`.
22//!
23//! # Example
24//!
25//! ```
26//! # use atomic_enum::atomic_enum;
27//! # use std::sync::atomic::Ordering;
28//! #[atomic_enum]
29//! #[derive(PartialEq)]
30//! enum CatState {
31//!     Dead = 0,
32//!     BothDeadAndAlive,
33//!     Alive,
34//! }
35//!
36//! let state = AtomicCatState::new(CatState::Dead);
37//! state.store(CatState::Alive, Ordering::Relaxed);
38//!
39//! assert_eq!(state.load(Ordering::Relaxed), CatState::Alive);
40//! ```
41//!
42//! This attribute does not use or generate any unsafe code.
43//!
44//! The crate can be used in a `#[no_std]` environment.
45

51fn enum_definition<'a>(
52    attrs: impl IntoIterator<Item = Attribute>,
53    vis: &Visibility,
54    ident: &Ident,
55    variants: impl IntoIterator<Item = &'a Variant>,
56) -> TokenStream2 {
57    let attrs = attrs.into_iter();
58    let variants = variants.into_iter();
59
60    quote! {
61        #(#attrs)*
62        #[derive(Debug, Clone, Copy)]
63        #vis enum #ident {
64            #( #variants ),*
65        }
66    }
67}

69fn atomic_enum_definition(vis: &Visibility, ident: &Ident, atomic_ident: &Ident) -> TokenStream2 {
70    let atomic_ident_docs = format!(
71        "A wrapper around [`{0}`] which can be safely shared between threads.
72
73This type uses an `AtomicUsize` to store the enum value.
74
75[`{0}`]: enum.{0}.html",
76        ident
77    );
78
79    quote! {
80        #[doc = #atomic_ident_docs]
81        #vis struct #atomic_ident(core::sync::atomic::AtomicUsize);
82    }
83}

85fn enum_to_usize(ident: &Ident) -> TokenStream2 {
86    quote! {
87        const fn to_usize(val: #ident) -> usize {
88            val as usize
89        }
90    }
91}

93fn enum_from_usize(ident: &Ident, variants: impl IntoIterator<Item = Variant>) -> TokenStream2 {
94    let variants_with_const_names: Vec<_> = variants
95        .into_iter()
96        .map(|v| v.ident)
97        .map(|id| {
98            let c_id = Ident::new(&format!("USIZE_{}", &id), id.span());
99            (id, c_id)
100        })
101        .collect();
102
103    let variant_consts = variants_with_const_names
104        .iter()
105        .map(|(id, c_id)| quote! { const #c_id: usize = #ident::#id as usize; });
106
107    let variants_back = variants_with_const_names
108        .iter()
109        .map(|(id, c_id)| quote! { #c_id => #ident::#id, });
110
111    quote! {
112        fn from_usize(val: usize) -> #ident {
113            #![allow(non_upper_case_globals)]
114            #(#variant_consts)*
115
116            match val {
117                #(#variants_back)*
118                _ => panic!("Invalid enum discriminant"),
119            }
120        }
121    }
122}

124fn atomic_enum_new(ident: &Ident, atomic_ident: &Ident) -> TokenStream2 {
125    let atomic_ident_docs = format!(
126        "Creates a new atomic [`{0}`].
127
128[`{0}`]: enum.{0}.html",
129        ident
130    );
131
132    quote! {
133        #[doc = #atomic_ident_docs]
134        pub const fn new(v: #ident) -> #atomic_ident {
135            #atomic_ident(core::sync::atomic::AtomicUsize::new(Self::to_usize(v)))
136        }
137    }
138}

140fn atomic_enum_into_inner(ident: &Ident) -> TokenStream2 {
141    quote! {
142        /// Consumes the atomic and returns the contained value.
143        ///
144        /// This is safe because passing self by value guarantees that no other threads are concurrently accessing the atomic data.
145        pub fn into_inner(self) -> #ident {
146            Self::from_usize(self.0.into_inner())
147        }
148    }
149}

151fn atomic_enum_set(ident: &Ident) -> TokenStream2 {

154        ///
155        /// This is safe because the mutable reference guarantees that no other threads are concurrently accessing the atomic data.

173fn atomic_enum_swap_mut(ident: &Ident) -> TokenStream2 {
174    quote! {
175        /// Stores a value into the atomic, returning the previous value, without performing an atomic operation.
176        ///
177        /// This is safe because the mutable reference guarantees that no other threads are concurrently accessing the atomic data.
178        pub fn swap_mut(&mut self, v: #ident) -> #ident {
179            let r = self.get();
180            self.set(v);
181            r
182        }
183    }

186fn atomic_enum_load(ident: &Ident) -> TokenStream2 {
187    quote! {
188        /// Loads a value from the atomic.
189        ///
190        /// `load` takes an `Ordering` argument which describes the memory ordering of this operation. Possible values are `SeqCst`, `Acquire` and `Relaxed`.
191        ///
192        /// # Panics
193        ///
194        /// Panics if order is `Release` or `AcqRel`.
195        pub fn load(&self, order: core::sync::atomic::Ordering) -> #ident {
196            Self::from_usize(self.0.load(order))
197        }
198    }

217fn atomic_enum_swap(ident: &Ident) -> TokenStream2 {
218    quote! {
219        /// Stores a value into the atomic, returning the previous value.
220        ///
221        /// `swap` takes an `Ordering` argument which describes the memory ordering of this operation.
222        /// All ordering modes are possible. Note that using `Acquire` makes the store part of this operation `Relaxed`,
223        /// and using `Release` makes the load part `Relaxed`.
224        pub fn swap(&self, val: #ident, order: core::sync::atomic::Ordering) -> #ident {
225            Self::from_usize(self.0.swap(Self::to_usize(val), order))
226        }
227    }

230#[cfg(feature = "cas")]
231fn atomic_enum_compare_and_swap(ident: &Ident) -> TokenStream2 {
232    quote! {
233        /// Stores a value into the atomic if the current value is the same as the `current` value.
234        ///
235        /// The return value is always the previous value. If it is equal to `current`, then the value was updated.
236        ///
237        /// `compare_and_swap` also takes an `Ordering` argument which describes the memory ordering of this operation.
238        /// Notice that even when using `AcqRel`, the operation might fail and hence just perform an `Acquire` load, but
239        /// not have `Release` semantics. Using `Acquire` makes the store part of this operation `Relaxed` if it happens,
240        /// and using `Release` makes the load part `Relaxed`.
241        #[allow(deprecated)]
242        #[deprecated(note = "Use `compare_exchange` or `compare_exchange_weak` instead")]
243        pub fn compare_and_swap(
244            &self,
245            current: #ident,
246            new: #ident,
247            order: core::sync::atomic::Ordering
248        ) -> #ident {
249            Self::from_usize(self.0.compare_and_swap(
250                Self::to_usize(current),
251                Self::to_usize(new),
252                order
253            ))
254        }
255    }
256}

258#[cfg(feature = "cas")]
259fn atomic_enum_compare_exchange(ident: &Ident) -> TokenStream2 {
260    quote! {
261        /// Stores a value into the atomic if the current value is the same as the `current` value.
262        ///
263        /// The return value is a result indicating whether the new value was written and containing the previous value.
264        /// On success this value is guaranteed to be equal to `current`.
265        ///
266        /// `compare_exchange` takes two `Ordering` arguments to describe the memory ordering of this operation. The first
267        /// describes the required ordering if the operation succeeds while the second describes the required ordering when
268        /// the operation fails. Using `Acquire` as success ordering makes the store part of this operation `Relaxed`, and
269        /// using `Release` makes the successful load `Relaxed`. The failure ordering can only be `SeqCst`, `Acquire` or
270        /// `Relaxed` and must be equivalent to or weaker than the success ordering.
271        pub fn compare_exchange(
272            &self,
273            current: #ident,
274            new: #ident,
275            success: core::sync::atomic::Ordering,
276            failure: core::sync::atomic::Ordering
277        ) -> core::result::Result<#ident, #ident> {
278            self.0
279                .compare_exchange(
280                    Self::to_usize(current),
281                    Self::to_usize(new),
282                    success,
283                    failure
284                )
285                .map(Self::from_usize)
286                .map_err(Self::from_usize)
287        }
288    }
289}

345#[proc_macro_attribute]
346/// Creates an atomic wrapper around a C-style enum.
347///
348/// The generated type is a wrapper around `AtomicUsize` that transparently
349/// converts between the stored integer and the enum type. This attribute
350/// also automatically derives the `Debug`, `Copy` and `Clone` traits on
351/// the enum type.
352///
353/// The name of the atomic type is the name of the enum type, prefixed with
354/// `Atomic`.
355///
356/// ```
357/// # use atomic_enum::atomic_enum;
358/// #[atomic_enum]
359/// enum State {
360///     On,
361///     Off,
362/// }
363///
364/// let state = AtomicState::new(State::Off);
365/// ```
366///
367/// The name can be overridden by passing an identifier
368/// as an argument to the attribute.
369///
370/// ```
371/// # use atomic_enum::atomic_enum;
372/// #[atomic_enum(StateAtomic)]
373/// enum State {
374///     On,
375///     Off,
376/// }
377///
378/// let state = StateAtomic::new(State::Off);
379/// ```
380pub fn atomic_enum(args: TokenStream, input: TokenStream) -> TokenStream {
381    // Parse the input
382    let ItemEnum {
383        attrs,
384        vis,
385        ident,
386        generics,
387        variants,
388        ..
389    } = parse_macro_input!(input as ItemEnum);
390
391    // We only support C-style enums: No generics, no fields
392    if !generics.params.is_empty() {
393        let span = generics.span();
394        let err = quote_spanned! {span=> compile_error!("Expected an enum without generics."); };
395        return err.into();
396    }
397
398    for variant in variants.iter() {
399        if !matches!(variant.fields, syn::Fields::Unit) {
400            let span = variant.fields.span();
401            let err =
402                quote_spanned! {span=> compile_error!("Expected a variant without fields."); };
403            return err.into();
404        }
405    }
406
407    // Define the enum
408    let mut output = enum_definition(attrs, &vis, &ident, &variants);
409
410    // Define the atomic wrapper
411    let atomic_ident = parse_macro_input!(args as Option<Ident>)
412        .unwrap_or_else(|| Ident::new(&format!("Atomic{}", ident), ident.span()));
413    output.extend(atomic_enum_definition(&vis, &ident, &atomic_ident));
414
415    // Write the impl block for the atomic wrapper
416    let enum_to_usize = enum_to_usize(&ident);
417    let enum_from_usize = enum_from_usize(&ident, variants);
418    let atomic_enum_new = atomic_enum_new(&ident, &atomic_ident);
419    let atomic_enum_into_inner = atomic_enum_into_inner(&ident);
420    let atomic_enum_set = atomic_enum_set(&ident);
421    let atomic_enum_get = atomic_enum_get(&ident);
422    let atomic_enum_swap_mut = atomic_enum_swap_mut(&ident);
423    let atomic_enum_load = atomic_enum_load(&ident);
424    let atomic_enum_store = atomic_enum_store(&ident);
425
426    output.extend(quote! {
427        impl #atomic_ident {
428            #enum_to_usize
429            #enum_from_usize
430            #atomic_enum_new
431            #atomic_enum_into_inner
432            #atomic_enum_set
433            #atomic_enum_get
434            #atomic_enum_swap_mut
435            #atomic_enum_load
436            #atomic_enum_store
437        }
438    });
439
440    #[cfg(feature = "cas")]
441    {
442        let atomic_enum_swap = atomic_enum_swap(&ident);
443        let atomic_enum_compare_and_swap = atomic_enum_compare_and_swap(&ident);
444        let atomic_enum_compare_exchange = atomic_enum_compare_exchange(&ident);
445        let atomic_enum_compare_exchange_weak = atomic_enum_compare_exchange_weak(&ident);
446
447        output.extend(quote! {
448            impl #atomic_ident {
449                #atomic_enum_swap
450                #atomic_enum_compare_and_swap
451                #atomic_enum_compare_exchange
452                #atomic_enum_compare_exchange_weak
453            }
454        });
455    }
456
457    // Implement the from and debug traits
458    output.extend(from_impl(&ident, &atomic_ident));
459    output.extend(debug_impl(&atomic_ident));
460
461    output.into()
462}
