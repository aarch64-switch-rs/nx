1//! Initial code/entrypoint support and utils
2//!
3//! # Custom entrypoint
4//!
5//! If you wish to define your custom entrypoint, you can do so by redefining the `__nx_rrt0_entry` weak fn.
6//!
7//! Example (check [here](https://switchbrew.org/wiki/Homebrew_ABI#Entrypoint_Arguments) for more entrypoint details):
8//! ```
9//! #[no_mangle]
10//! unsafe extern "C" fn __nx_rrt0_entry(arg0: usize, arg1: usize) {
11//!     // ...
12//! }
13//! ```
14//!
15//! # Custom version setup
16//!
17//! On the default entrypoint routine, the internal system version (see [`get_version`][`version::get_version`] and [`set_version`][`version::set_version`]) gets set the following way:
18//! * If the process was launched through HBL, use the "HOS version" value we got from it
19//! * Otherwise (and if using the `services` feature), use settings services ([`SystemSettingsServer`][`crate::service::set::ISystemSettingsServer`]) to get it
20//!
21//! If you wish to define your custom version setup (for instance, in contexts in which you wish to avoid executing the aforementioned setup), you can do so by redefining the `initialize_version` weak fn.
22//!
23//! Example:
24//! ```
25//! #[no_mangle]
26//! fn initialize_version(hbl_hos_version_opt: Option<hbl::Version>) {
27//!     // ...
28//! }
29//! ```

104/// Represents the process module format used by processes
105///
106/// This layout has to be present at the start of the process's `.rodata` section, containing its module name
107#[derive(Copy, Clone)]
108#[repr(C)]
109pub struct ModulePath {
110    /// Unused value
111    _zero: u32,
112    /// The length of the module name
113    path_len: u32,
114    /// The module name string
115    path: util::ArrayString<0x200>,
116}
117
118impl ModulePath {
119    /// Creates a [`ModulePath`] with the given module name
120    ///
121    /// # Arguments
122    ///
123    /// * `name`: The module name
124    #[inline]
125    pub const fn new(name: &str) -> Self {
126        Self {
127            _zero: 0,
128            path_len: util::const_usize_min(name.len(), 0x200 - 1) as u32,
129            path: util::ArrayString::from_str_truncate_null(name),
130        }
131    }
132
133    pub fn set_name(&mut self, new_name: &str) {
134        self.path = util::ArrayString::from_str(new_name);
135        self.path_len = new_name.len() as u32
136    }
137
138    pub fn get_name(&self) -> util::ArrayString<0x200> {
139        self.path
140    }
141}

157static G_MAIN_THREAD: sync::Mutex<Option<thread::Thread>> = sync::Mutex::new(None);
158static EH_FRAME_HDR_SECTION: elf::EhFrameHdrPtr = elf::EhFrameHdrPtr::new();

160/// Exits the current process
161///
162/// This will call the HBL-specific exit fn if running as a homebrew NRO, or [`exit_process`][`svc::exit_process`] otherwise
163pub fn exit(rc: ResultCode) -> ! {
164    match *G_EXIT_FN.lock() {
165        Some(exit_fn) => (exit_fn)(rc),
166        None => svc::exit_process(),
167    }
168}

170#[no_mangle]
171#[linkage = "weak"]
172fn initialize_version(hbl_hos_version_opt: Option<hbl::Version>) {
173    if let Some(hbl_hos_version) = hbl_hos_version_opt {
174        unsafe { version::set_version(hbl_hos_version.to_version()) };
175    } else {
176        #[cfg(feature = "services")]
177        {
178            use crate::ipc::sf::set::ISystemSettingsServer;
179            let set_sys = service::new_service_object::<set::SystemSettingsServer>().unwrap();
180            let mut fw_version: set::FirmwareVersion = Default::default();
181            set_sys
182                .get_firmware_version(sf::Buffer::from_mut_var(&mut fw_version))
183                .unwrap();
184
185            let version =
186                version::Version::new(fw_version.major, fw_version.minor, fw_version.micro);
187            unsafe { version::set_version(version) };
188        }
189    }
190}

194#[inline]
195unsafe fn set_main_thread_tlr(handle: svc::Handle) {
196    let tlr_raw = get_thread_local_region();
197    (*tlr_raw).nx_thread_vars.handle = handle;
198    (*tlr_raw).nx_thread_vars.magic = thread::imp::LibNxThreadVars::MAGIC;
199
200    MAIN_THREAD.__nx_thread.handle = handle;
201
202    (*tlr_raw).nx_thread_vars.thread_ref = addr_of_mut!(MAIN_THREAD);
203}

205#[allow(unsafe_op_in_unsafe_fn)]
206unsafe fn normal_entry(
207    maybe_abi_cfg_entries_ptr: *const hbl::AbiConfigEntry,
208    maybe_main_thread_handle: usize,
209    lr_exit_fn: Option<ExitFn>,
210) {
211    let exec_type =
212        match !maybe_abi_cfg_entries_ptr.is_null() && (maybe_main_thread_handle == usize::MAX) {
213            true => ExecutableType::Nro,
214            false => ExecutableType::Nso,
215        };
216    set_executable_type(exec_type);
217
218    let mut heap = util::PointerAndSize::new(ptr::null_mut(), crate::mem::alloc::HEAP_SIZE);
219    let mut main_thread_handle = maybe_main_thread_handle as svc::Handle;
220    let mut hos_version_opt: Option<hbl::Version> = None;
221
222    // If we are a NRO, parse the config entries hbloader sent us
223    if exec_type == ExecutableType::Nro {
224        let mut abi_entry = maybe_abi_cfg_entries_ptr;
225        loop {
226            match (*abi_entry).key {
227                hbl::AbiConfigEntryKey::EndOfList => {
228                    let loader_info_data = (*abi_entry).value[0] as *mut u8;
229                    let loader_info_data_len = (*abi_entry).value[1] as usize;
230                    if loader_info_data_len > 0 {
231                        let loader_info_slice =
232                            core::slice::from_raw_parts(loader_info_data, loader_info_data_len);
233                        if let Ok(loader_info) = core::str::from_utf8(loader_info_slice) {
234                            hbl::set_loader_info(loader_info);
235                        }
236                    }
237                    break;
238                }
239                hbl::AbiConfigEntryKey::MainThreadHandle => {
240                    main_thread_handle = (*abi_entry).value[0] as svc::Handle;
241                }
242                hbl::AbiConfigEntryKey::NextLoadPath => {
243                    // lengths from nx-hbloader:source/main.c
244                    // https://github.com/switchbrew/nx-hbloader/blob/cd6a723acbeabffd827a8bdc40563066f5401fb7/source/main.c#L13-L14
245                    let next_load_path: &'static mut util::ArrayString<512> =
246                        core::mem::transmute((*abi_entry).value[0]);
247                    let next_load_argv: &'static mut util::ArrayString<2048> =
248                        core::mem::transmute((*abi_entry).value[1]);
249                    hbl::set_next_load_entry_ptr(next_load_path, next_load_argv);
250                }
251                hbl::AbiConfigEntryKey::OverrideHeap => {
252                    heap.address = (*abi_entry).value[0] as *mut u8;
253                    heap.size = (*abi_entry).value[1] as usize;
254                }
255                hbl::AbiConfigEntryKey::OverrideService => {
256                    // todo!("OverrideService");
257                }
258                hbl::AbiConfigEntryKey::Argv => {
259                    // todo!("Argv");
260                }
261                hbl::AbiConfigEntryKey::SyscallAvailableHint => {
262                    // todo!("SyscallAvailableHint");
263                }
264                hbl::AbiConfigEntryKey::AppletType => {
265                    let applet_type: hbl::AppletType = mem::transmute((*abi_entry).value[0] as u32);
266                    hbl::set_applet_type(applet_type);
267                }
268                hbl::AbiConfigEntryKey::ProcessHandle => {
269                    let proc_handle = (*abi_entry).value[0] as Handle;
270                    hbl::set_process_handle(proc_handle);
271                }
272                hbl::AbiConfigEntryKey::LastLoadResult => {
273                    let last_load_rc = ResultCode::new((*abi_entry).value[0] as u32);
274                    hbl::set_last_load_result(last_load_rc);
275                }
276                hbl::AbiConfigEntryKey::RandomSeed => {
277                    let random_seed = ((*abi_entry).value[0], (*abi_entry).value[1]);
278                    hbl::set_random_seed(random_seed);
279                }
280                hbl::AbiConfigEntryKey::UserIdStorage => {
281                    // todo!("UserIdStorage");
282                }
283                hbl::AbiConfigEntryKey::HosVersion => {
284                    let hos_version_v = (*abi_entry).value[0] as u32;
285                    let os_impl_magic = (*abi_entry).value[1];
286                    hos_version_opt = Some(hbl::Version::new(hos_version_v, os_impl_magic));
287                }
288                _ => {
289                    // TODO: invalid config entries?
290                }
291            }
292            abi_entry = abi_entry.add(1);
293        }
294    }
295
296    if exec_type == ExecutableType::Nso {
297        // we need to set up our own ThreadLocalRegion with our thread structs
298        set_main_thread_tlr(main_thread_handle);
299    }
300
301    // Initialize virtual memory
302    vmem::initialize().unwrap();
303
304    // Set exit function (will be null for non-hbl NROs)
305    if exec_type == ExecutableType::Nro {
306        *G_EXIT_FN.lock() = lr_exit_fn;
307    }
308
309    // Initialize heap and memory allocation
310    heap = alloc::configure_heap(heap);
311    alloc::initialize(heap);
312
313    let main_thread_handle: thread::Thread =
314        thread::Thread::new_remote("MainThread", main_thread_handle);
315    G_MAIN_THREAD.set(Some(main_thread_handle));
316
317    // Initialize version support
318    initialize_version(hos_version_opt);
319
320    // Unwrap main(), which will trigger a panic if it didn't succeed
321    let res = main();
322
323    res.unwrap();
324
325    // unmount fs devices
326    #[cfg(feature = "fs")]
327    {
328        // clears any globally held fs dev handles
329        crate::fs::unmount_all();
330        // clears the global fsp_srv session.
331        crate::fs::finalize_fspsrv_session();
332    }
333
334    #[cfg(feature = "services")]
335    {
336        service::applet::finalize();
337
338        service::mii::finalize();
339    }
340
341    #[cfg(feature = "la")]
342    {
343        crate::la::finalize()
344    }
345
346    #[cfg(feature = "rand")]
347    {
348        crate::rand::finalize();
349    }
350
351    // Successful exit by default
352    exit(ResultSuccess::make());
353}

355#[no_mangle]
356#[linkage = "weak"]
357#[allow(unsafe_op_in_unsafe_fn)]
358unsafe extern "C" fn __nx_rrt0_entry(arg0: usize, arg1: usize) {
359    /*
360    Possible entry arguments:
361    - NSO/KIP: x0 = 0, x1 = <main-thread-handle>e
362    - NRO (hbl): x0 = <abi-config-entries-ptr>, x1 = usize::MAX
363    - Exception: x0 = <exception-type>, x1 = <stack-top>
364    */
365    if (arg0 != 0) && (arg1 != usize::MAX) {
366        //unreachable!("This should never happen as we handle it in the rrt0.s asm runtime");
367        // Handle exception entry
368        let exc_type: svc::ExceptionType = mem::transmute(arg0 as u32);
369        let stack_top = arg1 as *mut u8;
370        crate::exception::__nx_exception_dispatch(exc_type, stack_top);
371    }
372
373    let lr_raw: usize;
374    asm!(
375        "mov {}, lr",
376        out(reg) lr_raw
377    );
378
379    let lr_exit_fn: Option<ExitFn> = match lr_raw {
380        0 => None,
381        ptr => unsafe { Some(core::mem::transmute::<usize, ExitFn>(ptr)) },
382    };
383
384    // We actually want `_start` which is at the start of the .text region, but we don't know if
385    // it will be close enough to support lookup via `adr`.
386    // Since this function is in `.text` anyway, use QueryMemory SVC to find the actual start
387    let self_base_address: *const u8;
388    asm!(
389        "adr {}, __nx_rrt0_entry",
390        out(reg) self_base_address
391    );
392
393    // TODO: migrate to the providence APIs
394    let (info, _) = svc::query_memory(self_base_address).unwrap();
395    let aslr_base_address = info.base_address as *mut u8;
396
397    // assume that the MOD0 structure is at the start of .text
398    let mod0 = elf::mod0::Header::from_text_start_addr(aslr_base_address);
399    let start_dyn = mod0.get_dyn_start();
400    elf::relocate_with_dyn(aslr_base_address, start_dyn as *const elf::Dyn);
401
402    mod0.zero_bss_section();
403
404    let eh_hdr_ptr_start = mod0.get_eh_frame_header_start();
405    EH_FRAME_HDR_SECTION.set(eh_hdr_ptr_start);
406    let _ = unwinding::custom_eh_frame_finder::set_custom_eh_frame_finder(&EH_FRAME_HDR_SECTION);
407
408    // make sure that the writes are complete before there are any accesses
409    core::sync::atomic::compiler_fence(core::sync::atomic::Ordering::SeqCst);
410
411    // Handle NSO/KIP/NRO normal entry
412    let maybe_abi_cfg_entries_ptr = arg0 as *const hbl::AbiConfigEntry;
413    let maybe_main_thread_handle = arg1;
414    normal_entry(
415        maybe_abi_cfg_entries_ptr,
416        maybe_main_thread_handle,
417        lr_exit_fn,
418    );
