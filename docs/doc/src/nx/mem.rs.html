14/// Flushes data cache at a certain memory region
15///
16/// # Arguments
17///
18/// * `address`: Memory region address
19/// * `size`: Memory region size
20/// # Safety
21///
22/// Null pointers are OK as we are just doing cache invalidation, not accessing the pointer.
23#[inline(always)]
24#[allow(clippy::not_unsafe_ptr_arg_deref)]
25pub fn flush_data_cache(address: *mut u8, size: usize) {
26    extern "C" {
27        fn __nx_mem_flush_data_cache(address: *mut u8, size: usize);
28    }
29
30    unsafe {
31        __nx_mem_flush_data_cache(address, size);
32    }
33}

35/// Blocks thread until the memory region specified has the permission passed
36///
37/// # Arguments
38///
39/// * `address`: The address to query for memory permissions
40/// * `permissions`: The memory permission to wait on
41///
42/// Note that if multiple permissions are specified (e.g. `MemoryPermission::Read | MemoryPermission::Write`), the function will return if *any* specified permission is present.
43#[inline(always)]
44pub fn wait_for_permission(
45    address: svc::Address,
46    permission: svc::MemoryPermission,
47    timeout: Option<usize>,
48) -> crate::result::Result<()> {
49    let mut iteration: usize = 0;
50    loop {
51        let (memory, _) = svc::query_memory(address)?;
52        if memory.permission.contains(permission) {
53            return Ok(());
54        }
55        if timeout.is_some() && timeout <= Some(100_000 * iteration) {
56            // The timeout has been set and has already expired
57            return Err(svc::rc::ResultTimedOut::make());
58        }
59        iteration += 1;
60        let _ = crate::thread::sleep(100_000);
61    }
62}

64#[repr(transparent)]
65pub struct Shared<T: ?Sized> {
66    pub(crate) inner: Arc<Mutex<T>>,
67}
68impl<T> Shared<T> {
69    pub fn new(val: T) -> Self {
70        Self {
71            inner: Arc::new(Mutex::new(val)),
72        }
73    }
74}
75impl<T: ?Sized> Clone for Shared<T> {
76    fn clone(&self) -> Self {
77        Self {
78            inner: self.inner.clone(),
79        }
80    }
81}
82
83impl<T: ?Sized> Deref for Shared<T> {
84    type Target = Mutex<T>;
85    fn deref(&self) -> &Self::Target {
86        &self.inner
87    }
88}
