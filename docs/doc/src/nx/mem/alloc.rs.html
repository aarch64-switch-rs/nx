34/// Default implementation
35#[linkage = "weak"]
36#[unsafe(export_name = "__nx_mem_alloc_configure_heap")]
37pub fn configure_heap(heap_override: PointerAndSize) -> PointerAndSize {
38    if heap_override.is_valid() {
39        heap_override
40    } else {
41        let heap_size = match heap_override.size {
42            0 => {
43                let mem_available = svc::get_info(
44                    svc::InfoId::TotalMemorySize,
45                    svc::CURRENT_PROCESS_PSEUDO_HANDLE,
46                    0,
47                )
48                .unwrap();
49                let mem_used = svc::get_info(
50                    svc::InfoId::UsedMemorySize,
51                    svc::CURRENT_PROCESS_PSEUDO_HANDLE,
52                    0,
53                )
54                .unwrap();
55                if mem_available > mem_used + 0x200000 {
56                    ((mem_available - mem_used - 0x200000) & !0x1FFFFF) as usize
57                } else {
58                    (0x2000000 * 16) as usize
59                }
60            }
61            non_zero => non_zero,
62        };
63
64        let heap_addr = svc::set_heap_size(heap_size).unwrap();
65        debug_assert!(
66            !heap_addr.is_null(),
67            "Received null heap address after requesting from the kernel"
68        );
69
70        let (heap_metadata, _) = svc::query_memory(heap_addr).unwrap();
71        if !heap_metadata
72            .permission
73            .contains(svc::MemoryPermission::Read() | svc::MemoryPermission::Write())
74        {
75            unsafe {
76                svc::set_memory_permission(
77                    heap_addr,
78                    heap_size,
79                    svc::MemoryPermission::Read() | svc::MemoryPermission::Write(),
80                )
81                .unwrap();
82            }
83        }
84
85        PointerAndSize::new(heap_addr, heap_size)
86    }
87}

91/// Represents a heap allocator for this library
92/// # Safety
93///
94/// As with the regular Allocator trait, the `delete` function can only be called on pointers produced by the same implementation's `new`
95pub unsafe trait AllocatorEx: Allocator {
96    /// Allocates a new heap value
97    #[allow(clippy::new_ret_no_self)]
98    #[allow(clippy::wrong_self_convention)]
99    fn new<T>(&self) -> Result<NonNull<T>> {
100        let layout = Layout::new::<T>();
101        match self.allocate(layout) {
102            Ok(allocation) => Ok(allocation.cast()),
103            Err(_) => rc::ResultOutOfMemory::make_err(),
104        }
105    }
106
107    /// Releases a heap value
108    ///
109    /// The value must have been created using [`AllocatorEx::new`]
110    ///
111    /// # Arguments
112    ///
113    /// * `t`: Heap value address
114    fn delete<T>(&self, t: *mut T) {
115        let layout = Layout::new::<T>();
116        unsafe { self.deallocate(NonNull::new_unchecked(t.cast()), layout) };
117    }
118}

122#[global_allocator]
123static GLOBAL_ALLOCATOR: linked_list_allocator::LockedHeap =
124    linked_list_allocator::LockedHeap::empty();
125
126/// Initializes the global allocator with the given address and size.
127/// Returns a bool to indicate if the memory was consumed by the allocator
128///
129/// # Arguments
130///
131/// * `heap`: The heap address and size
132pub fn initialize(heap: PointerAndSize) -> bool {
133    unsafe { GLOBAL_ALLOCATOR.lock().init(heap.address, heap.size) };
134    false
135}
136
137/// Gets whether heap allocations are enabled
138///
139/// The library may internally disable them in exceptional cases: for instance, to avoid exception handlers to allocate heap memory if the exception cause is actually an OOM situation
140pub fn is_enabled() -> bool {
141    let alloc_guard = GLOBAL_ALLOCATOR.lock();
142    alloc_guard.size() != 0
143}

145/// Represents a wrapped and manually managed heap value
146///
147/// Note that a [`Buffer`] is able to hold both a single value or an array of values of the provided type
148pub struct Buffer<T, A: Allocator = Global> {
149    /// The actual heap value
150    pub ptr: *mut T,
151    /// The memory's layout
152    pub layout: Layout,
153    /// The allocator used to request the buffer
154    allocator: A,
155}
156
157impl<T> Buffer<T> {
158    /// Creates a new [`Buffer`] using the global allocator
159    ///
160    /// # Arguments
161    ///
162    /// * `align`: The align to use
163    /// * `count`: The count of values to allocate
164    pub fn new(align: usize, count: usize) -> Result<Self> {
165        let layout = Layout::from_size_align(count * mem::size_of::<T>(), align)
166            .map_err(|_| ResultCode::new(rc::ResultLayoutError::get_value()))?;
167        let allocator = Global;
168        let ptr = allocator.allocate(layout)?.as_ptr().cast();
169        Ok(Self {
170            ptr,
171            layout,
172            allocator,
173        })
174    }
175
176    /// Creates a new, invalid [`Buffer`]
177    #[inline]
178    pub const fn empty() -> Self {
179        Self {
180            ptr: ptr::null_mut(),
181            layout: Layout::new::<u8>(), // Dummy value
182            allocator: Global,
183        }
184    }
185}

187impl<T, A: Allocator> Buffer<T, A> {
188    /// Creates a new [`Buffer`] using a given allocator
189    ///
190    /// # Arguments
191    ///
192    /// * `align`: The align to use
193    /// * `count`: The count of values to allocate
194    /// * `allocator`: The allocator to use
195    pub fn new_in(align: usize, count: usize, allocator: A) -> Result<Self> {
196        let layout = Layout::from_size_align(count * mem::size_of::<T>(), align)
197            .map_err(|_| ResultCode::new(rc::ResultLayoutError::get_value()))?;
198        let ptr = allocator.allocate(layout)?.as_ptr().cast();
199        Ok(Self {
200            ptr,
201            layout,
202            allocator,
203        })
204    }
205
206    /// Gets whether this [`Buffer`] is valid
207    #[inline]
208    pub fn is_valid(&self) -> bool {
209        !self.ptr.is_null()
210    }
211
212    pub fn into_raw(value: Self) -> *mut [T] {
213        let no_drop = ManuallyDrop::new(value);
214        core::ptr::slice_from_raw_parts_mut(
215            no_drop.ptr,
216            no_drop.layout.size() / mem::size_of::<T>(),
217        )
218    }
219
220    /// Releases the [`Buffer`]
221    ///
222    /// The [`Buffer`] becomes invalid after this
223    ///
224    /// # Safety
225    ///
226    /// The buffer must never be read after this, as the internal buffer pointer is wiped. The buffer must also be for
227    pub unsafe fn release(&mut self) {
228        if self.is_valid() {
229            unsafe {
230                self.allocator
231                    .deallocate(NonNull::new_unchecked(self.ptr.cast()), self.layout);
232            }
233            self.ptr = core::ptr::null_mut();
234        }
235    }
236}

238impl<T> Index<usize> for Buffer<T> {
239    type Output = T;
240
241    fn index(&self, index: usize) -> &Self::Output {
242        assert!(self.is_valid(), "Indexing into invalid buffer");
243        assert!(
244            index <= (self.layout.size() / size_of::<T>()),
245            "Index out of bounds."
246        );
247
248        // SAFETY - we check that this pointer is valid via the two asserts above
249        unsafe { self.ptr.add(index).as_ref().unwrap_unchecked() }
250    }
251}
