1//! `LogManager` logger implementation

3use alloc::string::ToString;

5use super::*;
6use crate::ipc::sf;
7use crate::rrt0;
8use crate::service;
9use crate::service::lm::{self, ILoggerClient, ILoggingClient, LoggingService};
10use crate::svc;
11
12/// Represents a logger through [`LogService`][`lm::LoggingService`] services
13pub struct LmLogger {
14    logger: Option<lm::Logger>,
15}
16
17impl Logger for LmLogger {
18    fn new() -> Self {
19        Self {
20            logger: service::new_service_object::<LoggingService>()
21                .map(|mut log_srv| log_srv.open_logger(sf::ProcessId::new()).ok())
22                .ok()
23                .flatten(),
24        }
25    }
26
27    fn log(&mut self, metadata: &LogMetadata) {
28        if let Some(logger_obj) = self.logger.as_mut() {
29            let mut log_packet = logpacket::LogPacket::new();
30
31            if let Ok(process_id) = svc::get_process_id(svc::CURRENT_PROCESS_PSEUDO_HANDLE) {
32                log_packet.set_process_id(process_id);
33            }
34
35            let cur_thread = unsafe { thread::current().as_ref().unwrap() };
36            if let Ok(thread_id) = cur_thread.id() {
37                log_packet.set_thread_id(thread_id);
38            }
39
40            log_packet.set_file_name(String::from(metadata.file_name));
41            log_packet.set_function_name(String::from(metadata.fn_name));
42            log_packet.set_line_number(metadata.line_number);
43
44            let mod_name = rrt0::get_module_name()
45                .get_name()
46                .get_string()
47                .unwrap_or("aarch64-switch-rs (invalid module name)".to_string());
48            log_packet.set_module_name(mod_name);
49
50            log_packet.set_text_log(metadata.msg.clone());
51
52            let thread_name = cur_thread
53                .name
54                .get_str()
55                .unwrap_or("aarch64-switch-rs (invalid thread name)");
56            log_packet.set_thread_name(String::from(thread_name));
57
58            for packet in log_packet.encode_packet() {
59                let _ = logger_obj.log(sf::Buffer::from_array(&packet));
60            }
61        }
62    }
63}
