227    /// Updates the [`Context`] with the given data
228    ///
229    /// # Arguments
230    ///
231    /// * `data`: The data to update with
232    pub fn update<T>(&mut self, data: &[T]) {
233        let data_size = core::mem::size_of_val(data);
234        let data_start = data.as_ptr() as *const u8;
235        self.bits_consumed += (((self.buffered_size + data_size) / BLOCK_SIZE) * BLOCK_SIZE) * 8;
236        let mut data_offset: usize = 0;
237        let mut cur_size = data_size;
238
239        if self.buffered_size > 0 {
240            let needed = BLOCK_SIZE - self.buffered_size;
241
242            let copyable = needed.min(cur_size);
243            unsafe {
244                ptr::copy(
245                    data_start.add(data_offset),
246                    self.buf.as_mut_ptr().add(self.buffered_size),
247                    copyable,
248                );
249            }
250            data_offset += copyable;
251            cur_size -= copyable;
252
253            if self.buffered_size == BLOCK_SIZE {
254                unsafe {
255                    self.process_blocks(self.buf.as_ptr(), 1);
256                }
257                self.buffered_size = 0;
258            }
259        }
260
261        if cur_size >= BLOCK_SIZE {
262            let block_count = cur_size / BLOCK_SIZE;
263            unsafe {
264                self.process_blocks(data_start.add(data_offset), block_count);
265            }
266            let blocks_size = BLOCK_SIZE * block_count;
267            data_offset += blocks_size;
268            cur_size -= blocks_size;
269        }
270
271        if cur_size > 0 {
272            unsafe {
273                ptr::copy(data_start.add(data_offset), self.buf.as_mut_ptr(), cur_size);
274            }
275            self.buffered_size = cur_size;
276        }
277    }

279    /// Gets the produced hash (produces it first if not done yet)
280    ///
281    /// The output hash array must have size [`HASH_SIZE`] in bytes or this will fail with [`ResultInvalidSize`][`rc::ResultInvalidSize`]
282    ///
283    /// # Arguments
284    ///
285    /// * `out_hash`: The output array to fill with the hash
286    pub fn get_hash<T>(&mut self, out_hash: &mut [T]) -> Result<()> {
287        result_return_unless!(
288            core::mem::size_of_val(out_hash) == HASH_SIZE,
289            rc::ResultInvalidSize
290        );
291
292        if !self.finalized {
293            // Process last block, if necessary
294            self.bits_consumed += 8 * self.buffered_size;
295            self.buf[self.buffered_size] = 0x80;
296            self.buffered_size += 1;
297
298            let last_block_max_size = BLOCK_SIZE - mem::size_of::<u64>();
299            unsafe {
300                if self.buffered_size <= last_block_max_size {
301                    ptr::write_bytes(
302                        self.buf.as_mut_ptr().add(self.buffered_size),
303                        0,
304                        last_block_max_size - self.buffered_size,
305                    );
306                } else {
307                    ptr::write_bytes(
308                        self.buf.as_mut_ptr().add(self.buffered_size),
309                        0,
310                        BLOCK_SIZE - self.buffered_size,
311                    );
312                    self.process_blocks(self.buf.as_ptr(), 1);
313
314                    ptr::write_bytes(self.buf.as_mut_ptr(), 0, last_block_max_size);
315                }
316
317                let be_bits_consumed = self.bits_consumed.swap_bytes();
318                *(self.buf.as_mut_ptr().add(last_block_max_size) as *mut usize) = be_bits_consumed;
319                self.process_blocks(self.buf.as_ptr(), 1);
320            }
321            self.finalized = true;
322        }
323
324        unsafe {
325            let out_hash_buf_32 = out_hash.as_mut_ptr() as *mut u32;
326            for i in 0..HASH_SIZE_32 {
327                *out_hash_buf_32.add(i) = self.intermediate_hash[i].swap_bytes();
328            }
329        }
330
331        Ok(())
332    }

340/// Wrapper for directly calculating the hash of given data
341///
342/// The output hash array must have size [`HASH_SIZE`] in bytes or this will fail with [`ResultInvalidSize`][`rc::ResultInvalidSize`]
343///
344/// This essentially creates a [`Context`], updates it with the given data and produces its hash
345///
346/// # Arguments
347///
348/// * `data`: Input data
349/// * `out_hash`: Output array to fill into
350#[inline]
351pub fn calculate_hash<T, U>(data: &[T], out_hash: &mut [U]) -> Result<()> {
352    let mut ctx = Context::new();
353    ctx.update(data);
354    ctx.get_hash(out_hash)
