1//! Hardware-accelerated SHA-256 HMAC support
2
3use crate::crypto::{rc, sha256};
4use crate::result::*;
5use core::ptr;
6
7/// Represent the context used for SHA-256 HMAC operations
8pub struct Context {
9    sha_ctx: sha256::Context,
10    key: [u32; sha256::BLOCK_SIZE_32],
11    mac: [u32; sha256::HASH_SIZE_32],
12    finalized: bool,
13}
14
15impl Context {
16    /// Creates a new [`Context`]
17    ///
18    /// # Arguments
19    ///
20    /// * `key`: The key to use
21    pub fn new(key: &[u8]) -> Result<Self> {
22        let mut ctx = Self {
23            sha_ctx: sha256::Context::new(),
24            key: [0; sha256::BLOCK_SIZE_32],
25            mac: [0; sha256::HASH_SIZE_32],
26            finalized: false,
27        };
28
29        if key.len() <= sha256::BLOCK_SIZE {
30            unsafe {
31                ptr::copy(key.as_ptr(), ctx.key.as_mut_ptr() as *mut u8, key.len());
32            }
33        } else {
34            ctx.sha_ctx.update(key);
35            ctx.sha_ctx.get_hash(&mut ctx.key)?;
36        }
37
38        for i in 0..ctx.key.len() {
39            ctx.key[i] ^= super::IPAD_VAL;
40        }
41
42        ctx.sha_ctx.reset();
43        ctx.sha_ctx.update(&ctx.key);
44
45        Ok(ctx)
46    }
47
48    /// Updates the [`Context`] with the given data
49    ///
50    /// # Arguments
51    ///
52    /// * `data`: The data to update with
53    pub fn update<T>(&mut self, data: &[T]) {
54        self.sha_ctx.update(data);
55    }
56
57    /// Gets the output MAC
58    ///
59    /// The output hash array must have size [`HASH_SIZE`][`sha256::HASH_SIZE`] in bytes or this will fail with [`ResultInvalidSize`][`rc::ResultInvalidSize`]
60    ///
61    /// # Arguments
62    ///
63    /// * `out_mac`: Output array to fill into
64    pub fn get_mac<T>(&mut self, out_mac: &mut [T]) -> Result<()> {
65        result_return_unless!(
66            core::mem::size_of_val(out_mac) == sha256::HASH_SIZE,
67            rc::ResultInvalidSize
68        );
69
70        if !self.finalized {
71            self.sha_ctx.get_hash(&mut self.mac)?;
72
73            for i in 0..self.key.len() {
74                self.key[i] ^= super::IPAD_XOR_OPAD_VAL;
75            }
76
77            self.sha_ctx.reset();
78            self.sha_ctx.update(&self.key);
79            self.sha_ctx.update(&self.mac);
80            self.sha_ctx.get_hash(&mut self.mac)?;
81
82            self.finalized = true;
83        }
84
85        unsafe {
86            ptr::copy(
87                self.mac.as_ptr() as *const u8,
88                out_mac.as_mut_ptr() as *mut u8,
89                sha256::HASH_SIZE,
90            );
91        }
92
93        Ok(())
94    }
95}
96
97/// Wrapper for directly calculating the MAC of given data
98///
99/// The output hash array must have size [`HASH_SIZE`][`sha256::HASH_SIZE`] in bytes or this will fail with [`ResultInvalidSize`][`rc::ResultInvalidSize`]
100///
101/// This essentially creates a [`Context`], updates it with the given data and produces its MAC
102///
103/// # Arguments
104///
105/// * `key`: Input key
106/// * `data`: Input data
107/// * `out_,ac`: Output array to fill into
108#[inline]
109pub fn calculate_mac<T>(key: &[u8], data: &[u8], out_mac: &mut [T]) -> Result<()> {
110    let mut ctx = Context::new(key)?;
111    ctx.update(data);
112    ctx.get_mac(out_mac)
113}
